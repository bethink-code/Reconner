# Column Mapping Conflict Resolution UI

## Executive Summary

This document outlines the implementation of a smart conflict resolution UI for when users map multiple columns to the same field during file upload. The system automatically analyzes data quality and recommends the best column to use.

---

## Problem Statement

### Current UX Issue

When users upload files with similar column names (e.g., `_1` and `Date / Time` both containing date data), they can accidentally map both to the same field:

```
⚠ Warning: Both "_1" and "Date / Time" are mapped to Date

User sees:
[Dropdown: _1 → Date]
[Dropdown: Date / Time → Date]  ← Which one do I ignore?!

User thinks: "I have no idea which one has the actual data!"
```

**Result:** Users guess randomly, leading to:
- Invalid data being imported (header rows as data)
- Valid data being ignored
- Zero matches in reconciliation
- Confused and frustrated users

---

## Solution Overview

### Smart Conflict Resolution System

1. **Automatic Detection** - Detect duplicate mappings when user clicks "Confirm"
2. **Data Quality Analysis** - Analyze sample rows to determine which column has valid data
3. **Visual Comparison** - Show side-by-side comparison with recommendations
4. **One-Click Resolution** - User clicks button to choose the correct column

---

## Implementation Guide

### Step 1: Detect Mapping Conflicts

```typescript
interface MappingConflict {
  field: string;        // e.g., "date", "amount"
  columns: string[];    // e.g., ["_1", "Date / Time"]
}

function detectMappingConflicts(
  columnMapping: Record<string, string>
): MappingConflict[] {
  const fieldToColumns: Record<string, string[]> = {};
  
  // Group columns by their mapped field
  for (const [column, field] of Object.entries(columnMapping)) {
    if (field === 'ignore') continue;
    
    if (!fieldToColumns[field]) {
      fieldToColumns[field] = [];
    }
    fieldToColumns[field].push(column);
  }
  
  // Return only fields with multiple columns
  return Object.entries(fieldToColumns)
    .filter(([_, columns]) => columns.length > 1)
    .map(([field, columns]) => ({ field, columns }));
}

// Usage in your confirm handler:
function handleConfirmMapping() {
  const conflicts = detectMappingConflicts(columnMapping);
  
  if (conflicts.length > 0) {
    // Show conflict resolution modal
    setShowConflictModal(true);
    setConflicts(conflicts);
  } else {
    // Proceed with processing
    confirmMapping();
  }
}
```

### Step 2: Analyze Column Data Quality

```typescript
interface ColumnQuality {
  column: string;
  validCount: number;
  headerCount: number;
  emptyCount: number;
  recommendation: 'RECOMMENDED' | 'ACCEPTABLE' | 'NOT_RECOMMENDED';
  reason: string;
  sampleValues: string[];
}

function analyzeColumnQuality(
  column: string,
  sampleRows: any[],
  field: string
): ColumnQuality {
  const values = sampleRows.map(row => row[column]);
  
  let validCount = 0;
  let headerCount = 0;
  let emptyCount = 0;
  const sampleValues: string[] = [];
  
  for (const val of values) {
    const str = String(val || '').trim();
    
    // Collect first 3 samples
    if (sampleValues.length < 3) {
      sampleValues.push(str || '(empty)');
    }
    
    if (!str) {
      emptyCount++;
      continue;
    }
    
    // Check if it's a header keyword
    const headerKeywords = [
      'Date', 'Time', 'Date / Time', 'Amount', 
      'Reference', 'Description', 'Invoice', 'Card Number'
    ];
    if (headerKeywords.includes(str)) {
      headerCount++;
      continue;
    }
    
    // Check if it looks like valid data for this field
    let isValid = false;
    switch (field) {
      case 'date':
        // Excel serial: 45948.938...
        // ISO format: 2025-11-27
        // Formatted: 27/11/2025 or 27 Nov
        isValid = /^\d{4,5}/.test(str) || 
                  /^\d{4}-\d{2}-\d{2}/.test(str) || 
                  /^\d{1,2}[\/\-]\d{1,2}/.test(str) ||
                  /^\d{1,2}\s+\w{3}/.test(str);
        break;
        
      case 'time':
        // HH:MM:SS or HH:MM
        isValid = /^\d{1,2}:\d{2}(:\d{2})?$/.test(str);
        break;
        
      case 'amount':
        // Numeric with optional currency and comma separators
        isValid = /^[R$€£]?\s?[\d,]+\.?\d*$/.test(str);
        break;
        
      case 'reference':
      case 'description':
        // Any non-empty, non-header string
        isValid = str.length > 0;
        break;
        
      case 'cardNumber':
        // Card numbers: ****1234 or full numbers
        isValid = /^\*{4}\d{4}$/.test(str) || /^\d{12,19}$/.test(str);
        break;
        
      default:
        // For other fields, assume non-empty is valid
        isValid = true;
    }
    
    if (isValid) {
      validCount++;
    }
  }
  
  // Calculate recommendation
  const validPercent = (validCount / values.length) * 100;
  const headerPercent = (headerCount / values.length) * 100;
  
  let recommendation: ColumnQuality['recommendation'];
  let reason: string;
  
  if (validPercent > 80 && headerPercent < 10) {
    recommendation = 'RECOMMENDED';
    reason = `${validPercent.toFixed(0)}% valid data`;
  } else if (validPercent > 50) {
    recommendation = 'ACCEPTABLE';
    reason = `${validPercent.toFixed(0)}% valid data, ${headerCount} headers found`;
  } else {
    recommendation = 'NOT_RECOMMENDED';
    reason = `Only ${validPercent.toFixed(0)}% valid data, ${headerCount} headers found`;
  }
  
  return {
    column,
    validCount,
    headerCount,
    emptyCount,
    recommendation,
    reason,
    sampleValues
  };
}
```

### Step 3: Conflict Resolution Modal Component

```tsx
import React from 'react';
import {
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
  Button,
  Alert,
  Box,
  Typography,
  Grid,
  Card,
  CardContent,
  Chip,
  Table,
  TableBody,
  TableCell,
  TableContainer,
  TableHead,
  TableRow,
  Paper
} from '@mui/material';

interface ConflictResolutionModalProps {
  open: boolean;
  conflicts: MappingConflict[];
  sampleRows: any[];
  onResolve: (field: string, chosenColumn: string) => void;
  onCancel: () => void;
}

export function ConflictResolutionModal({
  open,
  conflicts,
  sampleRows,
  onResolve,
  onCancel
}: ConflictResolutionModalProps) {
  const [resolutions, setResolutions] = React.useState<Record<string, string>>({});
  
  // Track which conflicts have been resolved
  const allResolved = conflicts.every(c => resolutions[c.field]);
  
  const handleResolveAll = () => {
    // Apply all resolutions
    for (const [field, column] of Object.entries(resolutions)) {
      onResolve(field, column);
    }
  };
  
  return (
    <Dialog open={open} maxWidth="lg" fullWidth>
      <DialogTitle>
        Resolve Mapping Conflicts
      </DialogTitle>
      
      <DialogContent>
        <Alert severity="info" sx={{ mb: 3 }}>
          Multiple columns are mapped to the same field. 
          We've analyzed your data to help you choose the best column for each field.
        </Alert>
        
        {conflicts.map((conflict, index) => (
          <ConflictResolver
            key={conflict.field}
            conflict={conflict}
            sampleRows={sampleRows}
            isLast={index === conflicts.length - 1}
            selectedColumn={resolutions[conflict.field]}
            onSelect={(column) => {
              setResolutions(prev => ({ ...prev, [conflict.field]: column }));
            }}
          />
        ))}
      </DialogContent>
      
      <DialogActions>
        <Button onClick={onCancel} color="inherit">
          Cancel
        </Button>
        <Button 
          onClick={handleResolveAll}
          variant="contained"
          disabled={!allResolved}
        >
          Apply Changes
        </Button>
      </DialogActions>
    </Dialog>
  );
}

interface ConflictResolverProps {
  conflict: MappingConflict;
  sampleRows: any[];
  isLast: boolean;
  selectedColumn?: string;
  onSelect: (column: string) => void;
}

function ConflictResolver({
  conflict,
  sampleRows,
  isLast,
  selectedColumn,
  onSelect
}: ConflictResolverProps) {
  // Analyze each column
  const analyses = conflict.columns.map(col => 
    analyzeColumnQuality(col, sampleRows, conflict.field)
  );
  
  // Find the recommended column
  const recommended = analyses.find(a => a.recommendation === 'RECOMMENDED')?.column ||
                      analyses.sort((a, b) => b.validCount - a.validCount)[0]?.column;
  
  return (
    <Box sx={{ mb: isLast ? 0 : 4 }}>
      <Typography variant="h6" gutterBottom>
        {conflict.field.charAt(0).toUpperCase() + conflict.field.slice(1)} Field
      </Typography>
      
      <Grid container spacing={2}>
        {analyses.map(analysis => {
          const isRecommended = analysis.column === recommended;
          const isSelected = analysis.column === selectedColumn;
          
          return (
            <Grid item xs={12} md={6} key={analysis.column}>
              <Card 
                variant="outlined"
                sx={{
                  border: isSelected ? 2 : 1,
                  borderColor: isSelected 
                    ? 'primary.main' 
                    : isRecommended 
                      ? 'success.main' 
                      : 'divider',
                  bgcolor: isSelected ? 'action.selected' : 'background.paper',
                  cursor: 'pointer',
                  transition: 'all 0.2s',
                  '&:hover': {
                    boxShadow: 2
                  }
                }}
                onClick={() => onSelect(analysis.column)}
              >
                <CardContent>
                  {/* Column Name & Badge */}
                  <Box sx={{ display: 'flex', alignItems: 'center', mb: 2 }}>
                    <Typography variant="subtitle1" fontWeight="bold">
                      {analysis.column}
                    </Typography>
                    {isRecommended && (
                      <Chip 
                        label="Recommended" 
                        color="success" 
                        size="small" 
                        sx={{ ml: 1 }}
                      />
                    )}
                    {isSelected && (
                      <Chip 
                        label="Selected" 
                        color="primary" 
                        size="small" 
                        sx={{ ml: 1 }}
                      />
                    )}
                  </Box>
                  
                  {/* Sample Data */}
                  <Typography variant="body2" color="text.secondary" gutterBottom>
                    Sample data:
                  </Typography>
                  <Box 
                    sx={{ 
                      fontFamily: 'monospace', 
                      fontSize: '0.875rem',
                      bgcolor: 'grey.100',
                      p: 1.5,
                      borderRadius: 1,
                      mb: 2,
                      maxHeight: 80,
                      overflow: 'auto'
                    }}
                  >
                    {analysis.sampleValues.map((val, i) => (
                      <div key={i}>{val}</div>
                    ))}
                  </Box>
                  
                  {/* Statistics */}
                  <TableContainer component={Paper} variant="outlined" sx={{ mb: 2 }}>
                    <Table size="small">
                      <TableBody>
                        <TableRow>
                          <TableCell>Valid data</TableCell>
                          <TableCell align="right">
                            <strong>{analysis.validCount}</strong> / {sampleRows.length}
                          </TableCell>
                        </TableRow>
                        <TableRow>
                          <TableCell>Headers found</TableCell>
                          <TableCell align="right">
                            {analysis.headerCount}
                          </TableCell>
                        </TableRow>
                        <TableRow>
                          <TableCell>Empty values</TableCell>
                          <TableCell align="right">
                            {analysis.emptyCount}
                          </TableCell>
                        </TableRow>
                      </TableBody>
                    </Table>
                  </TableContainer>
                  
                  {/* Recommendation Reason */}
                  <Alert 
                    severity={
                      analysis.recommendation === 'RECOMMENDED' ? 'success' :
                      analysis.recommendation === 'ACCEPTABLE' ? 'info' : 'error'
                    }
                    sx={{ mb: 2 }}
                  >
                    {analysis.reason}
                  </Alert>
                  
                  {/* Action Button */}
                  <Button
                    fullWidth
                    variant={isSelected ? 'contained' : isRecommended ? 'contained' : 'outlined'}
                    color={isSelected ? 'primary' : isRecommended ? 'success' : 'inherit'}
                    onClick={(e) => {
                      e.stopPropagation();
                      onSelect(analysis.column);
                    }}
                  >
                    {isSelected ? 'Selected' : 'Use This Column'}
                  </Button>
                </CardContent>
              </Card>
            </Grid>
          );
        })}
      </Grid>
    </Box>
  );
}
```

### Step 4: Apply Resolution

```typescript
function resolveConflict(
  field: string,
  chosenColumn: string,
  columnMapping: Record<string, string>,
  setColumnMapping: (mapping: Record<string, string>) => void
) {
  setColumnMapping(prev => {
    const updated = { ...prev };
    
    // Find all columns currently mapped to this field
    const conflictingColumns = Object.keys(updated).filter(
      col => updated[col] === field
    );
    
    // Set chosen column to the field
    updated[chosenColumn] = field;
    
    // Set other conflicting columns to 'ignore'
    for (const col of conflictingColumns) {
      if (col !== chosenColumn) {
        updated[col] = 'ignore';
      }
    }
    
    return updated;
  });
}
```

### Step 5: Integration with Existing Mapping UI

```tsx
function ColumnMappingPage() {
  const [columnMapping, setColumnMapping] = useState<Record<string, string>>({});
  const [showConflictModal, setShowConflictModal] = useState(false);
  const [conflicts, setConflicts] = useState<MappingConflict[]>([]);
  const [sampleRows, setSampleRows] = useState<any[]>([]);
  
  const handleConfirmMapping = () => {
    // Detect conflicts
    const detectedConflicts = detectMappingConflicts(columnMapping);
    
    if (detectedConflicts.length > 0) {
      setConflicts(detectedConflicts);
      setShowConflictModal(true);
    } else {
      // No conflicts, proceed
      processFile();
    }
  };
  
  const handleResolveConflict = (field: string, chosenColumn: string) => {
    resolveConflict(field, chosenColumn, columnMapping, setColumnMapping);
  };
  
  const handleApplyResolutions = () => {
    setShowConflictModal(false);
    // Now proceed with processing
    processFile();
  };
  
  return (
    <>
      {/* Existing mapping UI */}
      <ColumnMappingForm
        columnMapping={columnMapping}
        onMappingChange={setColumnMapping}
      />
      
      <Button onClick={handleConfirmMapping}>
        Confirm Mapping
      </Button>
      
      {/* Conflict Resolution Modal */}
      <ConflictResolutionModal
        open={showConflictModal}
        conflicts={conflicts}
        sampleRows={sampleRows}
        onResolve={handleResolveConflict}
        onCancel={() => setShowConflictModal(false)}
      />
    </>
  );
}
```
