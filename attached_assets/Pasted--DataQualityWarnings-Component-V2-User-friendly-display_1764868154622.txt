 * DataQualityWarnings Component V2
 * 
 * User-friendly display of data quality issues with:
 * - Plain English explanations
 * - Clear indication of automatic vs manual actions
 * - Contextual help for resolving issues
 * - Reassuring tone that guides users through the process
 */

import React from 'react';
import {
  Alert,
  AlertTitle,
  Box,
  Button,
  Card,
  CardContent,
  Chip,
  Collapse,
  Divider,
  IconButton,
  Paper,
  Stack,
  Table,
  TableBody,
  TableCell,
  TableContainer,
  TableHead,
  TableRow,
  Typography
} from '@mui/material';
import {
  CheckCircle as CheckIcon,
  ExpandMore as ExpandMoreIcon,
  ExpandLess as ExpandLessIcon,
  AutoFixHigh as AutoFixIcon,
  Info as InfoIcon,
  Warning as WarningIcon,
  CleaningServices as CleanIcon,
  TableChart as TableIcon,
  Help as HelpIcon,
  ArrowForward as ArrowIcon
} from '@mui/icons-material';

// Types matching the validator output
interface DataQualityIssue {
  type: 'COLUMN_SHIFT' | 'PAGE_BREAK_ROWS' | 'REPEATED_HEADERS' | 'EMPTY_COLUMN' | 'DATA_TYPE_MISMATCH' | 'INCONSISTENT_DATA';
  severity: 'CRITICAL' | 'WARNING' | 'INFO';
  message: string;
  details: Record<string, any>;
  affectedRows?: number[];
  suggestedFix?: string;
}

interface ColumnAnalysis {
  columnName: string;
  columnIndex: number;
  inferredType: string;
  nullCount: number;
  nonNullCount: number;
  uniqueValues: number;
  sampleValues: string[];
  headerLikeValues: number;
  pageLikeValues: number;
}

interface DataQualityReport {
  hasIssues: boolean;
  hasCriticalIssues: boolean;
  totalRows: number;
  cleanRows: number;
  problematicRows: number;
  issues: DataQualityIssue[];
  columnAnalysis: ColumnAnalysis[];
  suggestedColumnMapping: Record<string, string>;
  rowsToRemove: number[];
  columnShiftDetected: boolean;
  shiftDetails?: any;
}

interface DataQualityWarningsProps {
  report: DataQualityReport;
  fileName: string;
  onContinue: () => void;
  onCancel?: () => void;
  onPreviewRows?: (rowIndices: number[]) => void;
}

// ============ USER-FRIENDLY EXPLANATIONS ============

interface IssueExplanation {
  title: string;
  whatHappened: string;
  whyItHappened: string;
  whatWeDo: string;
  userAction: 'none' | 'review' | 'required';
  userActionText?: string;
  icon: React.ReactNode;
  color: 'success' | 'info' | 'warning' | 'error';
}

function getIssueExplanation(issue: DataQualityIssue, report: DataQualityReport): IssueExplanation {
  switch (issue.type) {
    case 'PAGE_BREAK_ROWS':
      return {
        title: 'Page breaks detected',
        whatHappened: `Found ${issue.details.count?.toLocaleString() || 'some'} rows that contain page numbers or print headers instead of actual data.`,
        whyItHappened: 'This happens when a file is exported from a paginated report or printed view. The page numbers (like "Page 1 of 10") get included as data rows.',
        whatWeDo: "We'll automatically remove these rows before processing. Your actual transaction data won't be affected.",
        userAction: 'none',
        icon: <CleanIcon />,
        color: 'success'
      };

    case 'REPEATED_HEADERS':
      const pattern = issue.details.pattern;
      return {
        title: 'Repeated headers in data',
        whatHappened: `Found ${issue.details.count?.toLocaleString() || 'some'} rows where the column headers appear again in the middle of the data.`,
        whyItHappened: pattern 
          ? `This file was exported from a paginated view. ${pattern}.`
          : 'This happens when Excel repeats headers on each printed page, and those get saved into the export.',
        whatWeDo: "We'll automatically remove these duplicate header rows. Only your actual data will be processed.",
        userAction: 'none',
        icon: <CleanIcon />,
        color: 'success'
      };

    case 'COLUMN_SHIFT':
      return {
        title: 'Column headers don\'t match the data',
        whatHappened: 'The data in some columns doesn\'t match what the column header says it should be.',
        whyItHappened: 'This can happen when columns get shifted during export, or when the source system uses generic column names. For example, a column named "Description" might actually contain quantity values.',
        whatWeDo: 'We\'ve analyzed the actual data and will suggest the correct mapping.',
        userAction: 'review',
        userActionText: 'When mapping columns in the next step, pay attention to the sample data shown - don\'t rely on header names alone.',
        icon: <WarningIcon />,
        color: 'warning'
      };

    case 'EMPTY_COLUMN':
      return {
        title: 'Empty columns found',
        whatHappened: `Found ${issue.details.columns?.length || 'some'} columns that are empty or mostly empty.`,
        whyItHappened: 'These columns may be unused in your export, or the data might be in different columns than expected.',
        whatWeDo: 'Empty columns will be available but you can skip them during mapping.',
        userAction: 'none',
        userActionText: 'You can ignore these columns when setting up your mapping.',
        icon: <InfoIcon />,
        color: 'info'
      };

    case 'DATA_TYPE_MISMATCH':
      // Extract column name from message if available
      const colMatch = issue.message.match(/Column "([^"]+)"/);
      const colName = colMatch ? colMatch[1] : 'Some columns';
      return {
        title: `${colName} contains unexpected data`,
        whatHappened: issue.message,
        whyItHappened: 'This usually means the column contains a mix of data types, or the header name doesn\'t match the actual content.',
        whatWeDo: 'We\'ll show you sample values from each column so you can map them correctly.',
        userAction: 'review',
        userActionText: 'Check the sample data when mapping this column to make sure it contains what you expect.',
        icon: <TableIcon />,
        color: 'warning'
      };

    default:
      return {
        title: issue.message,
        whatHappened: issue.message,
        whyItHappened: 'An unexpected data quality issue was detected.',
        whatWeDo: issue.suggestedFix || 'Please review before continuing.',
        userAction: 'review',
        icon: <WarningIcon />,
        color: 'warning'
      };
  }
}

// ============ COMPONENTS ============

function IssueCard({ 
  issue, 
  report,
  onPreviewRows 
}: { 
  issue: DataQualityIssue; 
  report: DataQualityReport;
  onPreviewRows?: (rows: number[]) => void;
}) {
  const [expanded, setExpanded] = React.useState(false);
  const explanation = getIssueExplanation(issue, report);

  const actionBadge = {
    none: { label: '✓ Auto-fixed', color: 'success' as const },
    review: { label: 'Review recommended', color: 'warning' as const },
    required: { label: 'Action required', color: 'error' as const }
  }[explanation.userAction];

  return (
    <Card 
      variant="outlined" 
      sx={{ 
        mb: 2,
        borderColor: explanation.color === 'success' ? 'success.light' : 
                     explanation.color === 'warning' ? 'warning.light' : 
                     explanation.color === 'error' ? 'error.light' : 'divider',
        borderLeftWidth: 4,
        borderLeftStyle: 'solid',
        borderLeftColor: `${explanation.color}.main`
      }}
    >
      <CardContent sx={{ pb: expanded ? 2 : '16px !important' }}>
        {/* Header Row */}
        <Box 
          sx={{ 
            display: 'flex', 
            alignItems: 'flex-start', 
            justifyContent: 'space-between',
            cursor: 'pointer'
          }}
          onClick={() => setExpanded(!expanded)}
        >
          <Stack direction="row" spacing={2} alignItems="flex-start" sx={{ flex: 1 }}>
            <Box sx={{ 
              color: `${explanation.color}.main`,
              mt: 0.5
            }}>
              {explanation.icon}
            </Box>
            <Box sx={{ flex: 1 }}>
              <Stack direction="row" spacing={1} alignItems="center" flexWrap="wrap" sx={{ mb: 0.5 }}>
                <Typography variant="subtitle1" fontWeight={600}>
                  {explanation.title}
                </Typography>
                <Chip 
                  label={actionBadge.label}
                  size="small"
                  color={actionBadge.color}
                  variant={explanation.userAction === 'none' ? 'filled' : 'outlined'}
                />
              </Stack>
              <Typography variant="body2" color="text.secondary">
                {explanation.whatHappened}
              </Typography>
            </Box>
          </Stack>
          <IconButton size="small" sx={{ ml: 1 }}>
            {expanded ? <ExpandLessIcon /> : <ExpandMoreIcon />}
          </IconButton>
        </Box>

        {/* Expanded Details */}
        <Collapse in={expanded}>
          <Box sx={{ mt: 2, ml: 5 }}>
            {/* Why it happened */}
            <Paper 
              elevation={0} 
              sx={{ 
                p: 2, 
                mb: 2, 
                bgcolor: 'grey.50',
                borderRadius: 2
              }}
            >
              <Stack direction="row" spacing={1} alignItems="flex-start">
                <HelpIcon fontSize="small" sx={{ color: 'text.secondary', mt: 0.3 }} />
                <Box>
                  <Typography variant="caption" color="text.secondary" fontWeight={600} display="block">
                    Why did this happen?
                  </Typography>
                  <Typography variant="body2">
                    {explanation.whyItHappened}
                  </Typography>
                </Box>
              </Stack>
            </Paper>

            {/* What we'll do */}
            <Stack direction="row" spacing={1} alignItems="flex-start" sx={{ mb: 2 }}>
              <AutoFixIcon fontSize="small" sx={{ color: 'success.main', mt: 0.3 }} />
              <Box>
                <Typography variant="caption" color="text.secondary" fontWeight={600} display="block">
                  What we'll do
                </Typography>
                <Typography variant="body2">
                  {explanation.whatWeDo}
                </Typography>
              </Box>
            </Stack>

            {/* User action (if needed) */}
            {explanation.userAction !== 'none' && explanation.userActionText && (
              <Alert 
                severity={explanation.userAction === 'required' ? 'warning' : 'info'}
                sx={{ mt: 2 }}
              >
                <Typography variant="body2">
                  <strong>Your action:</strong> {explanation.userActionText}
                </Typography>
              </Alert>
            )}

            {/* Preview button */}
            {issue.affectedRows && issue.affectedRows.length > 0 && onPreviewRows && (
              <Button
                size="small"
                variant="outlined"
                onClick={(e) => {
                  e.stopPropagation();
                  onPreviewRows(issue.affectedRows!);
                }}
                sx={{ mt: 2 }}
              >
                Preview {Math.min(issue.affectedRows.length, 10)} affected rows
              </Button>
            )}
          </Box>
        </Collapse>
      </CardContent>
    </Card>
  );
}

function SummaryCard({ report }: { report: DataQualityReport }) {
  const autoFixCount = report.issues.filter(i => 
    i.type === 'PAGE_BREAK_ROWS' || i.type === 'REPEATED_HEADERS'
  ).length;
  
  const reviewCount = report.issues.filter(i => 
    i.type === 'COLUMN_SHIFT' || i.type === 'DATA_TYPE_MISMATCH'
  ).length;

  return (
    <Paper 
      elevation={0}
      sx={{ 
        p: 3, 
        mb: 3, 
        bgcolor: report.hasCriticalIssues ? 'warning.50' : 'success.50',
        border: '1px solid',
        borderColor: report.hasCriticalIssues ? 'warning.200' : 'success.200',
        borderRadius: 2
      }}
    >
      <Stack direction="row" spacing={3} alignItems="center" justifyContent="space-between" flexWrap="wrap">
        {/* Left side - summary text */}
        <Box>
          <Typography variant="h6" gutterBottom>
            {report.hasCriticalIssues 
              ? '⚠️ A few things need your attention'
              : '✅ Your file looks good!'}
          </Typography>
          <Typography variant="body2" color="text.secondary">
            {report.problematicRows > 0 
              ? `We'll automatically clean ${report.problematicRows.toLocaleString()} rows (${((report.problematicRows / report.totalRows) * 100).toFixed(1)}%) from your file.`
              : 'No cleanup needed.'}
          </Typography>
        </Box>

        {/* Right side - stats */}
        <Stack direction="row" spacing={3}>
          <Box sx={{ textAlign: 'center' }}>
            <Typography variant="h4" color="success.main" fontWeight={700}>
              {report.cleanRows.toLocaleString()}
            </Typography>
            <Typography variant="caption" color="text.secondary">
              Clean rows
            </Typography>
          </Box>
          {autoFixCount > 0 && (
            <Box sx={{ textAlign: 'center' }}>
              <Typography variant="h4" color="success.main" fontWeight={700}>
                {autoFixCount}
              </Typography>
              <Typography variant="caption" color="text.secondary">
                Auto-fixed
              </Typography>
            </Box>
          )}
          {reviewCount > 0 && (
            <Box sx={{ textAlign: 'center' }}>
              <Typography variant="h4" color="warning.main" fontWeight={700}>
                {reviewCount}
              </Typography>
              <Typography variant="caption" color="text.secondary">
                To review
              </Typography>
            </Box>
          )}
        </Stack>
      </Stack>
    </Paper>
  );
}

function ColumnMappingTip({ report }: { report: DataQualityReport }) {
  if (!report.columnShiftDetected) return null;

  return (
    <Alert 
      severity="info" 
      sx={{ mb: 3 }}
      icon={<TableIcon />}
    >
      <AlertTitle>Tip for Column Mapping</AlertTitle>
      <Typography variant="body2" paragraph>
        Some column headers in this file don't match their actual content. 
        When you map columns in the next step:
      </Typography>
      <Box component="ul" sx={{ m: 0, pl: 2 }}>
        <li>
          <Typography variant="body2">
            <strong>Look at the sample data</strong> shown for each column, not just the header name
          </Typography>
        </li>
        <li>
          <Typography variant="body2">
            <strong>Use the suggested mappings</strong> - we've analyzed the actual data to recommend the right columns
          </Typography>
        </li>
      </Box>
      
      {Object.keys(report.suggestedColumnMapping).length > 0 && (
        <Box sx={{ mt: 2 }}>
          <Typography variant="caption" color="text.secondary" display="block" gutterBottom>
            Suggested mappings based on data analysis:
          </Typography>
          <Stack direction="row" spacing={1} flexWrap="wrap" useFlexGap>
            {Object.entries(report.suggestedColumnMapping).slice(0, 5).map(([field, column]) => (
              <Chip
                key={field}
                label={`${field} → "${column}"`}
                size="small"
                variant="outlined"
                sx={{ mb: 0.5 }}
              />
            ))}
          </Stack>
        </Box>
      )}
    </Alert>
  );
}

// ============ MAIN COMPONENT ============

export function DataQualityWarnings({
  report,
  fileName,
  onContinue,
  onCancel,
  onPreviewRows
}: DataQualityWarningsProps) {
  
  // No issues - simple success message
  if (!report.hasIssues) {
    return (
      <Box sx={{ py: 4 }}>
        <Alert 
          severity="success" 
          icon={<CheckIcon fontSize="large" />}
          sx={{ mb: 3 }}
        >
          <AlertTitle>File validated successfully!</AlertTitle>
          <Typography variant="body2">
            <strong>{fileName}</strong> contains {report.totalRows.toLocaleString()} rows ready for processing.
            No data quality issues were detected.
          </Typography>
        </Alert>
        
        <Box sx={{ display: 'flex', justifyContent: 'flex-end', gap: 2 }}>
          {onCancel && (
            <Button variant="outlined" onClick={onCancel}>
              Cancel
            </Button>
          )}
          <Button 
            variant="contained" 
            onClick={onContinue}
            endIcon={<ArrowIcon />}
          >
            Continue to Column Mapping
          </Button>
        </Box>
      </Box>
    );
  }

  // Group issues by action type
  const autoFixIssues = report.issues.filter(i => 
    i.type === 'PAGE_BREAK_ROWS' || i.type === 'REPEATED_HEADERS' || i.type === 'EMPTY_COLUMN'
  );
  const reviewIssues = report.issues.filter(i => 
    i.type === 'COLUMN_SHIFT' || i.type === 'DATA_TYPE_MISMATCH' || i.type === 'INCONSISTENT_DATA'
  );

  return (
    <Box sx={{ py: 2 }}>
      {/* Header */}
      <Typography variant="h6" gutterBottom>
        Data Quality Check
      </Typography>
      <Typography variant="body2" color="text.secondary" sx={{ mb: 3 }}>
        We've analyzed <strong>{fileName}</strong> and found a few things to address.
      </Typography>

      {/* Summary Card */}
      <SummaryCard report={report} />

      {/* Column Mapping Tip (if shift detected) */}
      <ColumnMappingTip report={report} />

      {/* Issues that will be auto-fixed */}
      {autoFixIssues.length > 0 && (
        <Box sx={{ mb: 3 }}>
          <Typography variant="subtitle2" color="text.secondary" sx={{ mb: 1.5, display: 'flex', alignItems: 'center', gap: 1 }}>
            <CleanIcon fontSize="small" />
            Will be fixed automatically
          </Typography>
          {autoFixIssues.map((issue, index) => (
            <IssueCard 
              key={`auto-${index}`} 
              issue={issue} 
              report={report}
              onPreviewRows={onPreviewRows}
            />
          ))}
        </Box>
      )}

      {/* Issues to review */}
      {reviewIssues.length > 0 && (
        <Box sx={{ mb: 3 }}>
          <Typography variant="subtitle2" color="text.secondary" sx={{ mb: 1.5, display: 'flex', alignItems: 'center', gap: 1 }}>
            <WarningIcon fontSize="small" />
            Please review
          </Typography>
          {reviewIssues.map((issue, index) => (
            <IssueCard 
              key={`review-${index}`} 
              issue={issue} 
              report={report}
              onPreviewRows={onPreviewRows}
            />
          ))}
        </Box>
      )}

      {/* Action Buttons */}
      <Divider sx={{ my: 3 }} />
      
      <Box sx={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
        <Typography variant="body2" color="text.secondary">
          {report.cleanRows.toLocaleString()} rows will be processed
        </Typography>
        
        <Stack direction="row" spacing={2}>
          {onCancel && (
            <Button variant="outlined" onClick={onCancel}>
              Cancel
            </Button>
          )}
          <Button 
            variant="contained" 
            onClick={onContinue}
            endIcon={<ArrowIcon />}
            color={report.hasCriticalIssues ? 'warning' : 'primary'}
          >
            {report.problematicRows > 0 
              ? `Continue with ${report.cleanRows.toLocaleString()} Clean Rows`
              : 'Continue to Column Mapping'}
          </Button>
        </Stack>
      </Box>
    </Box>
  );
}

export default DataQualityWarnings;
