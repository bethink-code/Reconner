# Emergency Performance Fix Plan
## Pieter's Pomp Stasie Reconner - Report Page Hanging Issue

**Status:** 🔴 CRITICAL - Reports are unusable  
**Impact:** Users cannot view reconciliation results  
**Root Cause:** Loading 25,000+ transactions on every page load  
**Priority:** P0 - Fix immediately

---

## 🚨 The Problem

### Current Behavior
```
User clicks "View Report"
  ↓
Frontend: GET /api/periods/:id/transactions (no pagination)
  ↓
Backend: SELECT * FROM transactions WHERE period_id = '...' (returns 25,947 rows)
  ↓
Network: Transfers ~5-10 MB of JSON
  ↓
Frontend: Renders 25,947 rows in table
  ↓
Browser: Hangs for 10-30 seconds (or crashes)
```

### Why It's Happening

1. **No Pagination** - API returns ALL transactions at once
2. **No Lazy Loading** - Frontend tries to render everything immediately
3. **Repeated Calls** - Same data fetched multiple times per page
4. **Heavy Calculations** - Report metrics recalculated on every render
5. **DOM Overload** - Browser struggles with 25,000+ DOM nodes

---

### Fix 1.1: Add Report Summary Endpoint (30 min)

**Problem:** Report page calculates everything from 25,947 rows  
**Solution:** Pre-calculate summary metrics

```typescript
// NEW ENDPOINT: GET /api/periods/:periodId/summary
router.get('/periods/:periodId/summary', async (req, res) => {
  const { periodId } = req.params;
  
  // Single query with aggregations - FAST!
  const summary = await db.query(`
    SELECT 
      -- Total counts
      COUNT(*) as total_transactions,
      COUNT(DISTINCT CASE WHEN source_type = 'bank' THEN id END) as bank_count,
      COUNT(DISTINCT CASE WHEN source_type = 'fuel' THEN id END) as fuel_count,
      
      -- Match status counts
      COUNT(CASE WHEN match_status = 'matched' THEN 1 END) as matched_count,
      COUNT(CASE WHEN match_status = 'unmatched' THEN 1 END) as unmatched_count,
      COUNT(CASE WHEN match_status = 'partial' THEN 1 END) as partial_count,
      
      -- Amounts
      SUM(amount) as total_amount,
      SUM(CASE WHEN source_type = 'bank' THEN amount ELSE 0 END) as bank_total,
      SUM(CASE WHEN source_type = 'fuel' THEN amount ELSE 0 END) as fuel_total,
      
      -- Card/Cash breakdown (fuel only)
      COUNT(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'yes' THEN 1 END) as fuel_card_count,
      SUM(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'yes' THEN amount ELSE 0 END) as fuel_card_total,
      COUNT(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'no' THEN 1 END) as fuel_cash_count,
      SUM(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'no' THEN amount ELSE 0 END) as fuel_cash_total,
      COUNT(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'unknown' THEN 1 END) as fuel_unknown_count,
      SUM(CASE WHEN source_type = 'fuel' AND is_card_transaction = 'unknown' THEN amount ELSE 0 END) as fuel_unknown_total
      
    FROM transactions
    WHERE period_id = $1
  `, [periodId]);
  
  const row = summary.rows[0];
  
  // Calculate derived metrics
  const bankMatchRate = row.bank_count > 0 
    ? (row.matched_count / row.bank_count) * 100 
    : 0;
  
  const fuelCardMatchRate = row.fuel_card_count > 0
    ? (row.matched_count / row.fuel_card_count) * 100
    : 0;
  
  res.json({
    totalTransactions: parseInt(row.total_transactions),
    bankTransactions: parseInt(row.bank_count),
    fuelTransactions: parseInt(row.fuel_count),
    
    matched: parseInt(row.matched_count),
    unmatched: parseInt(row.unmatched_count),
    partial: parseInt(row.partial_count),
    
    totalAmount: parseFloat(row.total_amount || 0),
    bankTotal: parseFloat(row.bank_total || 0),
    fuelTotal: parseFloat(row.fuel_total || 0),
    
    fuelBreakdown: {
      card: {
        count: parseInt(row.fuel_card_count),
        total: parseFloat(row.fuel_card_total || 0)
      },
      cash: {
        count: parseInt(row.fuel_cash_count),
        total: parseFloat(row.fuel_cash_total || 0)
      },
      unknown: {
        count: parseInt(row.fuel_unknown_count),
        total: parseFloat(row.fuel_unknown_total || 0)
      }
    },
    
    matchRates: {
      bank: bankMatchRate,
      fuelCard: fuelCardMatchRate
    },
    
    discrepancy: parseFloat(row.fuel_card_total || 0) - parseFloat(row.bank_total || 0)
  });
});
```

### Fix 1.2: Add Basic Pagination (45 min)

**Update the transactions endpoint:**

```typescript
// UPDATED: GET /api/periods/:periodId/transactions
router.get('/periods/:periodId/transactions', async (req, res) => {
  const { periodId } = req.params;
  const page = parseInt(req.query.page as string) || 1;
  const limit = parseInt(req.query.limit as string) || 100;
  const offset = (page - 1) * limit;
  
  // Optional filters
  const matchStatus = req.query.matchStatus as string; // 'matched', 'unmatched', 'partial'
  const sourceType = req.query.sourceType as string;   // 'bank', 'fuel'
  const isCardTransaction = req.query.isCardTransaction as string;
  
  // Build WHERE clause
  let whereConditions = ['period_id = $1'];
  let params: any[] = [periodId];
  let paramIndex = 2;
  
  if (matchStatus) {
    whereConditions.push(`match_status = $${paramIndex}`);
    params.push(matchStatus);
    paramIndex++;
  }
  
  if (sourceType) {
    whereConditions.push(`source_type = $${paramIndex}`);
    params.push(sourceType);
    paramIndex++;
  }
  
  if (isCardTransaction) {
    whereConditions.push(`is_card_transaction = $${paramIndex}`);
    params.push(isCardTransaction);
    paramIndex++;
  }
  
  const whereClause = whereConditions.join(' AND ');
  
  // Get total count (for pagination info)
  const countResult = await db.query(
    `SELECT COUNT(*) as total FROM transactions WHERE ${whereClause}`,
    params
  );
  const total = parseInt(countResult.rows[0].total);
  
  // Get paginated results
  const result = await db.query(
    `SELECT 
      id,
      transaction_date,
      transaction_time,
      amount,
      description,
      reference_number,
      card_number,
      source_type,
      source_name,
      payment_type,
      is_card_transaction,
      match_status,
      match_id
    FROM transactions 
    WHERE ${whereClause}
    ORDER BY transaction_date DESC, transaction_time DESC
    LIMIT $${paramIndex} OFFSET $${paramIndex + 1}`,
    [...params, limit, offset]
  );
  
  res.json({
    data: result.rows,
    pagination: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
      hasMore: offset + limit < total
    }
  });
});
```

**Frontend Change:**
```tsx

    ['transactions', periodId, page, filters],

  return (

          ))}
        </TableBody>

      <TablePagination
        count={data.pagination.total}
        page={page - 1}
        rowsPerPage={100}
        onPageChange={(_, newPage) => setPage(newPage + 1)}
      />
    </>
  );
}
```

**Impact:** Only loads 100 rows at a time instead of 25,947

### Fix 2.3: Optimize Match Processing Breakdown (1 hour)

**Problem:** "Match Processing Breakdown" section recalculates from all transactions

**Solution:** Add to summary query:

```sql
-- Add to summary endpoint
SELECT 
  -- Existing fields...
  
  -- Match delay buckets
  COUNT(CASE WHEN m.date_difference_in_days = 0 THEN 1 END) as same_day_matches,
  COUNT(CASE WHEN m.date_difference_in_days = 1 THEN 1 END) as one_day_matches,
  COUNT(CASE WHEN m.date_difference_in_days = 2 THEN 1 END) as two_day_matches,
  COUNT(CASE WHEN m.date_difference_in_days >= 3 THEN 1 END) as three_plus_day_matches
  
FROM transactions t
LEFT JOIN matches m ON t.match_id = m.id
WHERE t.period_id = $1
```

## 🚀 PHASE 3: Long-Term Fixes (Next Week)

### Fix 3.1: Materialized Summary Table (3 hours)

**Store pre-calculated metrics in database:**

```sql
CREATE TABLE period_summaries (
  period_id UUID PRIMARY KEY REFERENCES reconciliation_periods(id),
  
  -- Counts
  total_transactions INTEGER,
  bank_count INTEGER,
  fuel_count INTEGER,
  matched_count INTEGER,
  unmatched_count INTEGER,
  
  -- Amounts
  total_amount DECIMAL(15,2),
  bank_total DECIMAL(15,2),
  fuel_total DECIMAL(15,2),
  
  -- Breakdown
  fuel_card_count INTEGER,
  fuel_card_total DECIMAL(15,2),
  fuel_cash_count INTEGER,
  fuel_cash_total DECIMAL(15,2),
  
  -- Match delays
  same_day_matches INTEGER,
  one_day_matches INTEGER,
  two_day_matches INTEGER,
  three_plus_day_matches INTEGER,
  
  -- Metadata
  last_calculated TIMESTAMP DEFAULT NOW(),
  
  CONSTRAINT check_counts CHECK (
    total_transactions = bank_count + fuel_count
  )
);

-- Update trigger: recalculate when transactions change
CREATE OR REPLACE FUNCTION update_period_summary()
RETURNS TRIGGER AS $$
BEGIN
  -- Recalculate summary for affected period
  INSERT INTO period_summaries (period_id, ...)
  SELECT 
    NEW.period_id,
    COUNT(*) as total_transactions,
    -- ... all the aggregations from Fix 1.1
  FROM transactions
  WHERE period_id = NEW.period_id
  ON CONFLICT (period_id) DO UPDATE SET
    total_transactions = EXCLUDED.total_transactions,
    -- ... update all fields
    last_calculated = NOW();
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_summary
AFTER INSERT OR UPDATE OR DELETE ON transactions
FOR EACH ROW
EXECUTE FUNCTION update_period_summary();
```

```typescript
// src/api/reconciliation.ts

export async function getSummary(periodId: string) {
  const response = await fetch(`/api/periods/${periodId}/summary`);
  return response.json();
}

export async function getTransactions(
  periodId: string, 
  options: {
    page?: number;
    limit?: number;
    matchStatus?: string;
    sourceType?: string;
  } = {}
) {
  const params = new URLSearchParams();
  if (options.page) params.set('page', String(options.page));
  if (options.limit) params.set('limit', String(options.limit));
  if (options.matchStatus) params.set('matchStatus', options.matchStatus);
  if (options.sourceType) params.set('sourceType', options.sourceType);
  
  const response = await fetch(
    `/api/periods/${periodId}/transactions?${params}`
  );
  return response.json();
}
```
